public class Program
{
	public static void Main()
	{//2-Crie um programa que leia um número do teclado até que encontre um número igual a zero. No final, mostre a soma dos números digitados.

     int num;
     int somador = 0;
     do 
     {
         Console.Write("Qual o seu numero: ");
         num = Convert.ToInt32(Console.ReadLine());
         somador += num;
     }while (num != 0);
     Console.WriteLine("A soma dos numeros é: " + somador );
     Console.ReadKey();
 }
}
